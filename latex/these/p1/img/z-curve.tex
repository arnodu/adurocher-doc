\tikzstyle{dot}=[circle,fill,inner sep=1.5]	
\newcommand*\zcurveone[1]{ 
	\begin{tikzpicture}[scale=#1]
	
	\tikzstyle{zpath}=[thick,color=black!60]
	
	\node [dot] (A00) at (0,0) {};
	\node [dot] (A01) at (1,0) {};
	\node [dot] (A10) at (0,1) {};
	\node [dot] (A11) at (1,1) {};
	
	\draw [zpath] (A00) -- (A10) -- (A01) -- (A11);
	
	\draw [dashed,color=black!50] (-0.5,0.5) -- (1.5,0.5);
	\draw [dashed,color=black!50] (0.5,-0.5) -- (0.5,1.5);
	\draw [color=black!50](-0.5,-0.5) rectangle (1.5,1.5);
	
	\end{tikzpicture}
}
\newcommand*\zcurvetwo[1]{ 
	\begin{tikzpicture}[scale=#1]
	
	\tikzstyle{zpath}=[thick,color=black!80]
	
	\node (B00) at (0,0) {\zcurveone{#1/2}};
	\node (B01) at (0,1) {\zcurveone{#1/2}};
	\node (B10) at (1,0) {\zcurveone{#1/2}};
	\node (B11) at (1,1) {\zcurveone{#1/2}};
	
	\draw [zpath] (-0.25,0.75) -- (0.25,0.25);
	\draw [zpath] (0.25,1.25) -- (0.75,-0.25);
	\draw [zpath] (0.75,0.75) -- (1.25,0.25);
	
	\draw [color=black](-0.5,-0.5) rectangle (1.5,1.5);
	\end{tikzpicture}
}
